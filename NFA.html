module NFA (Σ : Alphabet) where

record NFA (S : Type) {{_ : DecEq S}} : Type where
  field
    Q  : Set⟨ S ⟩
    -- Σ  : Alphabet
    δ  : S → Token → Set⟨ S ⟩
    Q₀ : ∃ (_∈′ Q)
    -- q₀ : S
    -- q₀∈
    F  : ∃ (_⊆′ Q)

  q₀ = proj₁ Q₀
  fs = proj₁ F

  isFinal : S → Type
  isFinal = _∈′ fs

  isFinal? : Decidable¹ isFinal
  isFinal? = _∈′? fs

  Δ : S → Word → Set⟨ S ⟩
  Δ s ε       = singleton s
  Δ s (α ∷ w) = ⋃ (flip Δ w) (δ s α)

  δ̂ : Word → Set⟨ S ⟩
  δ̂ = Δ q₀

open NFA public

instance
  L-NFA : ∀ {S} {{_ : DecEq S}} → Language (NFA S)
  L-NFA .accept nfa@(record {F = F , _}) w =
    (δ̂ nfa w ∩ F) ≢ ∅
